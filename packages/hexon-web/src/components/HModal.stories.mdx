<Canvas>
  <Story name="基础">
    {() => ({
      data() {
        return {
          show: false,
        };
      },

        showModal() {
          this.show = true;
        },
        hideModal() {
          this.show = false;
        },
      },
      components: { HModal, HButton },
      template: `
<HButton @click="showModal">show</HButton>
<HModal name="HModal" :show="show">
  <HButton @click="hideModal">hide</HButton>
</HModal>
`,
    })}
